<script>
  const playbookForm = document.getElementById('playbookForm');
  const terminalOutput = document.getElementById('terminalOutput');

  playbookForm.addEventListener('submit', function(event) {
    event.preventDefault();

    const playbook = document.getElementById('playbook').value;

    terminalOutput.innerHTML = '';

    const xhr = new XMLHttpRequest();
    xhr.open('POST', '/run_playbook', true);
    xhr.setRequestHeader('Content-Type', 'application/x-www-form-urlencoded');

    xhr.onreadystatechange = function() {
      if (xhr.readyState === XMLHttpRequest.DONE) {
        if (xhr.status === 200) {
          const output = xhr.responseText.trim();
          simulateTerminalOutput(output);
        } else {
          terminalOutput.innerHTML += '<div>Error occurred: ' + xhr.statusText + '</div>';
        }
      }
    };

    xhr.send('playbook=' + encodeURIComponent(playbook));
  });

  function simulateTerminalOutput(output) {
    const lines = output.split('\n');
    let delay = 0;

    lines.forEach(function(line) {
      setTimeout(function() {
        terminalOutput.innerHTML += '<div>' + line + '</div>';
        terminalOutput.scrollTop = terminalOutput.scrollHeight;
      }, delay);
      delay += 100;
    });
  }
</script>
